    Implement cleanup helpers (per D-11, D-12, Pitfall 2, Pitfall 8):
    ```go
    func (s *Session) cleanupAbortedTask(itt uint32) {
        // Per Pitfall 2: Unregister FIRST, then cancel, then remove
        s.router.Unregister(itt)
        s.mu.Lock()
        tk, ok := s.tasks[itt]
        if ok {
            delete(s.tasks, itt)
        }
        s.mu.Unlock()
        if ok {
            tk.cancel(ErrTaskAborted)
        }
    }

    func (s *Session) cleanupTasksByLUN(lun uint64) {
        // Per Pitfall 8: snapshot matching ITTs first, then clean each one
        s.mu.Lock()
        var toClean []uint32
        for itt, tk := range s.tasks {
            if tk.lun == lun {   // Need to add lun field to task struct
                toClean = append(toClean, itt)
            }
        }
        s.mu.Unlock()

        for _, itt := range toClean {
            s.cleanupAbortedTask(itt)
        }
    }
    ```

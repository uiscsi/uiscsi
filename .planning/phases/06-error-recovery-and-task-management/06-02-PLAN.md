<objective>
Implement ERL 0 (Error Recovery Level 0): automatic session reinstatement after connection failure with transparent in-flight command retry.

Purpose: ERL 0 is the foundation of iSCSI error recovery. When a connection drops, the library auto-reconnects, reinstates the session with same ISID+TSIH per RFC 7143 Section 6.3.5, and retries all in-flight commands so callers see transparent recovery.

Output: `internal/login/login.go` (WithTSIH), `internal/session/recovery.go`, `internal/session/recovery_test.go`, updated `session.go` and `async.go`
</objective>

        s.mu.Lock()
        if s.recovering {

    ```go
    func (s *Session) reconnect(cause error) {
        s.cfg.logger.Info("session: connection lost, starting ERL 0 recovery", "cause", cause)

        // Step 1: Stop old pumps by closing old connection (Pitfall 6).
        // Closing net.Conn unblocks pending Read/Write in pump goroutines.
        s.cancel() // cancel old context
        oldConn := s.conn
        oldConn.Close()

        // Step 2: Snapshot in-flight tasks.
        s.mu.Lock()
        taskSnapshot := make(map[uint32]*task, len(s.tasks))
        for itt, tk := range s.tasks {
            taskSnapshot[itt] = tk
        }
        // Clear tasks and router for fresh start
        s.tasks = make(map[uint32]*task)
        s.mu.Unlock()
        // Unregister all old ITTs from router
        for itt := range taskSnapshot {
            s.router.Unregister(itt)
        }

        // Step 3: Exponential backoff reconnect loop.
        var newParams *login.NegotiatedParams
        var newConn *transport.Conn
        var lastErr error

        for attempt := 0; attempt < s.cfg.maxReconnectAttempts; attempt++ {
            if attempt > 0 {
                delay := s.cfg.reconnectBackoff * time.Duration(1<<uint(attempt-1))
                time.Sleep(delay)
            }

            ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)

            // Dial new TCP connection
            tc, err := transport.Dial(ctx, s.targetAddr)
            if err != nil {
                cancel()
                lastErr = err
                s.cfg.logger.Warn("session: reconnect dial failed", "attempt", attempt+1, "error", err)
                continue
            }

            // Login with same ISID + old TSIH for session reinstatement (D-03)
            loginOpts := append([]login.LoginOption{
                login.WithISID(s.isid),
                login.WithTSIH(s.tsih),
            }, s.loginOpts...)

            params, err := login.Login(ctx, tc, loginOpts...)
            if err != nil {
                cancel()
                tc.Close()
                lastErr = err
                s.cfg.logger.Warn("session: reconnect login failed", "attempt", attempt+1, "error", err)
                continue
            }

            cancel()
            newConn = tc
            newParams = params
            break
        }

        if newConn == nil {
            // All attempts exhausted (D-02)
            s.mu.Lock()
            s.err = fmt.Errorf("session: reconnect failed after %d attempts: %w", s.cfg.maxReconnectAttempts, lastErr)
            s.recovering = false
            s.mu.Unlock()
            // Fail all snapshotted tasks
            for _, tk := range taskSnapshot {
                tk.cancel(fmt.Errorf("session: reconnect failed: %w", lastErr))
            }
            return
        }

        // Step 4: Replace session internals with new connection.
        newCtx, newCancel := context.WithCancel(context.Background())
        newWriteCh := make(chan *transport.RawPDU, 64)
        newUnsolCh := make(chan *transport.RawPDU, 16)

        s.mu.Lock()
        s.conn = newConn
        s.params = *newParams
        s.tsih = newParams.TSIH
        s.writeCh = newWriteCh
        s.unsolCh = newUnsolCh
        s.cancel = newCancel
        s.window = newCmdWindow(newParams.CmdSN, newParams.CmdSN, newParams.CmdSN)
        s.expStatSN = newParams.ExpStatSN
        s.err = nil
        s.loggedIn = true
        s.mu.Unlock()

        // Step 5: Start new pump goroutines.
        go s.readPumpLoop(newCtx)
        go s.writePumpLoop(newCtx)
        go s.dispatchLoop(newCtx)
        go s.keepaliveLoop(newCtx)

        // Step 6: Retry snapshotted tasks (D-04).
        s.retryTasks(newCtx, taskSnapshot)

        // Step 7: Mark recovery complete.
        s.mu.Lock()
        s.recovering = false
        s.mu.Unlock()

        s.cfg.logger.Info("session: ERL 0 recovery complete")
    }
    ```

    func (s *Session) retryTasks(ctx context.Context, tasks map[uint32]*task) {
        for _, tk := range tasks {
            // For write tasks, check if Reader is seekable (Pitfall 1)
            if tk.isWrite && tk.reader != nil {
                if seeker, ok := tk.reader.(io.Seeker); ok {
                    if _, err := seeker.Seek(0, io.SeekStart); err != nil {
                        tk.cancel(fmt.Errorf("session: retry seek failed: %w", err))
                        continue
                    }
                    tk.bytesSent = 0 // Reset byte tracking
                } else {
                    tk.cancel(ErrRetryNotPossible)
                    continue
                }
            }

            // Acquire new CmdSN (Pitfall 5 -- must use new window)
            cmdSN, err := s.window.acquire(ctx)
            if err != nil {
                tk.cancel(fmt.Errorf("session: retry acquire CmdSN: %w", err))
                continue
            }

            // Re-register task with router
            newITT := s.router.AllocateITT()
            pduCh := s.router.RegisterPersistent(newITT)

            // Build fresh SCSICommand PDU with new sequence numbers
            // Reuse original CDB and parameters from the task
            // ... (construct PDU, send via writeCh, start taskLoop)

            s.mu.Lock()
            tk.itt = newITT
            tk.nextDataSN = 0
            tk.nextOffset = 0
            if tk.isRead && tk.buf != nil {

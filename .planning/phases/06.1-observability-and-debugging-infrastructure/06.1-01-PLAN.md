    Add a helper function:
    ```go
    func truncHex(b []byte, maxBytes int) string {
        if len(b) > maxBytes {
            b = b[:maxBytes]
        }
        if len(b) == 0 {
            return ""
        }
        // Build "NN:NN:NN:..." format
        var buf strings.Builder
        for i, v := range b {
            if i > 0 {
                buf.WriteByte(':')
            }
            fmt.Fprintf(&buf, "%02x", v)
        }
        return buf.String()
    }
    ```
